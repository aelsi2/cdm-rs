1//! Startup code and minimal runtime for the CDM-16 processor.
2//!
3//! # Features
4//!
5//! This crate provides:
6//! - Default interrupt and exception handlers using the `halt` instruction
7//! - A macro for defining the entry point of the program: [`entry`](macro@entry)
8//! - A macro for redefining exception handlers: [`exception`](macro@exception)
9//! - A macro for defining interrupt handlers: [`interrupt`](macro@interrupt)
10//! - A macro for registering interrupt handlers in the IVT:
11//! [`interrupt_vectors`](macro@interrupt_vectors)
12//!
13//! # Cargo features
14//! #### `interrupts`
15//! This feature disables the default interrupt handlers and allows the use of [`interrupt_vectors`](macro@interrupt_vectors)
16//!
17//! # Requirements
18//! #### `memory.x`
19//! The crate expects a file named `memory.x` to be present in the project root directory.
20//! This file is a linker script that needs to specify the address range that the linker can use.
21//! This is done using the `MEMORY` command defining a region named `RAM`. The origin
22//! address must be greater than or equal to `0x100` so that it doesn't intersect with the IVT.
23//!
24//! You can reserve a range of addresses between the IVT and the main memory for MMIO by specifying a
25//! larger origin address for `RAM` (e.g `0x120`).
26//! ***
27//!
28//! #### Rust flags
29//! The crate generates a linker script named `link.x` in the output directory. It needs to be passed to the
30//! linker. This can be done by passing `-Clink-arg=-Tlink.x` to rustc.
31//! ***
32//!
33//! #### Entry point
34//! Exactly one function needs to be marked as the application entry point by applying the [`entry`](macro@entry) attribute.
35//!
36//! # Example
37//! `.cargo/config.toml`
38//! ```toml
39//! [build]
40//! target = "cdm-none"
41//! rustflags = [ "-Clink-arg=-Tlink.x" ]
42//! [unstable]
43//! build-std = [ "core" ]
44//! ```
45//!
46//! `memory.x`
47//! ```ld
48//! MEMORY {
49//!     RAM : ORIGIN = 0x100, LENGTH = 64K-0x100
50//! }
51//! ```
52//!
53//! `src/main.rs`
54//! ```rust
55//! #![no_std]
56//! #![no_main]
57//!
58//! use cdm_rt::entry;
59//!
60//! #[entry]
61//! fn main() -> ! {
62//!     loop { /* .. */ }
63//! }
64//! ```

87/// Represents a vector in the interrupt vector table.
88///
89/// The first field is the pointer to the handler function.
90/// The second field is the initial value of the processor status register.
91///
92/// Use `#[interrupt]` to define interrupt handler functions.
93#[derive(Clone, Copy)]
94#[repr(C)]
95pub struct InterruptVector(pub unsafe extern "cdm-isr" fn(), pub Psr);
96
97impl InterruptVector {
98    /// The default vector used in absence of an explicit definition.
99    ///
100    /// Calls `InterruptHandler`, which triggers a hardware halt by default.
101    pub const DEFAULT: Self = {
102        unsafe extern "cdm-isr" {
103            fn InterruptHandler();
104        }
105        InterruptVector(InterruptHandler, Psr::None)
106    };
107}

109/// Defines the application-specific interrupt handler section of the interrupt vector table.
110///
111/// The interrupt vectors specified in the arguments are placed sequentially after the reset and
112/// exception vectors, starting from index `INTERRUPT_START`. The rest of the table is filled
113/// with `InterruptVector::DEFAULT`.
114///
115/// Must be used **once** in the dependency graph.
116///
117/// Use the [`interrupt`](macro@interrupt) attribute to define interrupt handler functions.
118///
119/// ``` no_run
120/// interrupt_vectors![
121///     InterruptVector(MyHandler1, Psr::None), // int INTERRUPT_START+0
122///     InterruptVector(MyHandler2, Psr::None), // int INTERRUPT_START+1
123///     InterruptVector(MyHandler3, Psr::None), // int INTERRUPT_START+2
124/// ];
125///
126/// #[cdm_rt::interrupt]
127/// fn MyHandler1() { /* .. */ }
128/// #[cdm_rt::interrupt]
129/// fn MyHandler2() { /* .. */ }
130/// #[cdm_rt::interrupt]
131/// fn MyHandler3() { /* .. */ }
132/// ```
133#[cfg(feature = "interrupts")]
134#[macro_export]
135macro_rules! interrupt_vectors {
136    ($($elems:expr),* $(,)?) => {
137        #[used]
138        #[unsafe(no_mangle)]
139        #[unsafe(link_section = ".ivt.interrupts")]
140        static __INTERRUPTS: [$crate::InterruptVector; $crate::INTERRUPT_COUNT] = {
141            const fn make_array<const N: usize, const M: usize>(
142                prefix: [$crate::InterruptVector; M],
143            ) -> [$crate::InterruptVector; N] {
144                assert!(M <= N, "Prefix length cannot exceed array length");
145                let mut arr = [$crate::InterruptVector::DEFAULT; N];
146                let mut i: usize = 0;
147                while i < M {
148                    arr[i] = prefix[i];
149                    i += 1;
150                }
151                arr
152            }
153
154            make_array([$($elems),*])
155        };
156    };
157}

165    ".section .text._start",
166    ".global _start",
167    ".type _start,%function",
168    "_start:",
169    "ldi fp, 0",
170    "stsp fp",
171    "jsr main",
172    "halt",
